332def getAngle(a,b,c): 

333 ab = [b[0]-a[0],b[1]-a[1],b[2]-a[2]] 

334 bc = [c[0]-b[0],c[1]-b[1],c[2]-b[2]] 

335 

336 vectorAB = math.sqrt(ab[0] * ab[0] + ab[1] * ab[1] + ab[2] * ab[2]) 

337 vectorBC = math.sqrt(bc[0] * bc[0] + bc[1] * bc[1] + bc[2] * bc[2]) 

338 

339 abNorm = [ab[0] / vectorAB, ab[1] / vectorAB, ab[2] / vectorAB] 

340 bcNorm = [bc[0] / vectorBC, bc[1] / vectorBC, bc[2] / vectorBC] 

341 

342 res = abNorm[0] * bcNorm[0] + abNorm[1] * bcNorm[1] + abNorm[2] * bcNorm[2] 

343 

344 return math.acos(res)*180.0/ 3.141592653589793
